<script>
  const smsContainer = document.querySelector('.sms-container');
  const numStars = 50; // Number of stars

  for (let i = 0; i < numStars; i++) {
    const star = document.createElement('div');
    star.classList.add('star');
    const size = Math.random() * 3 + 1; // Star size between 1px and 4px
    star.style.width = `${size}px`;
    star.style.height = `${size}px`;

    // Position stars randomly around the border area
    const angle = Math.random() * Math.PI * 2;
    const radius = Math.min(smsContainer.offsetWidth, smsContainer.offsetHeight) / 2 + (Math.random() * 30 - 15); // Adjust radius for border area
    const x = smsContainer.offsetWidth / 2 + radius * Math.cos(angle) - size / 2;
    const y = smsContainer.offsetHeight / 2 + radius * Math.sin(angle) - size / 2;

    star.style.left = `${x}px`;
    star.style.top = `${y}px`;
    star.style.animationDelay = `${Math.random() * 2}s`;
    smsContainer.appendChild(star);
  }
</script>
